🔗 Integração com Google Sheets + Apps Script
Configure abaixo a URL do seu Google Apps Script (GAS) para conectar o sistema ao banco de dados em nuvem. Siga os passos para publicar seu Apps Script como Web App.
1. URL do Google Apps Script (Web App)
2. Código do Google Apps Script (cole no editor GAS)

Acesse script.google.com → Novo projeto → cole o código abaixo → Publicar como Web App (acesso: Qualquer pessoa).

// ============================================================ // SGQ Mercedes-Benz – Google Apps Script // Cole este código em: script.google.com > Novo Projeto // ============================================================ const SS_ID = 'SEU_ID_DA_PLANILHA'; let result; if (action === 'saveAQ') result = saveRow(ss, 'AQ', params.data); else if (action === 'saveBrainstorm') result = saveRow(ss, 'Brainstorm', params.data); else if (action === 'saveIshikawa') result = saveRow(ss, 'Ishikawa', params.data); else if (action === 'save5Por') result = saveRow(ss, '5Porques', params.data); else if (action === 'savePlano') result = saveRow(ss, 'PlanoAcao', params.data); else if (action === 'saveAuditoria') result = saveRow(ss, 'Auditorias', params.data); else if (action === 'saveChecklist') result = saveRow(ss, 'Checklist', params.data); else if (action === 'savePOP') result = saveRow(ss, 'POP', params.data); else if (action === 'saveProcesso') result = saveRow(ss, 'Processos', params.data); return ContentService.createTextOutput(JSON.stringify(result)) .setMimeType(ContentService.MimeType.JSON); } function doGet(e) { const sheet = e.parameter.sheet; const ss = SpreadsheetApp.openById(SS_ID); const sh = ss.getSheetByName(sheet); if (!sh) return json({data: []}); const rows = sh.getDataRange().getValues(); if (rows.length < 2) return json({data: []}); const headers = rows[0]; const data = rows.slice(1).map(row => { let obj = {}; headers.forEach((h, i) => obj[h] = row[i]); return obj; }); return json({data}); } function saveRow(ss, sheetName, data) { let sh = ss.getSheetByName(sheetName); if (!sh) sh = ss.insertSheet(sheetName); if (sh.getLastRow() === 0) sh.appendRow(Object.keys(data)); sh.appendRow(Object.values(data)); return {success: true}; } function json(obj) { return ContentService.createTextOutput(JSON.stringify(obj)) .setMimeType(ContentService.MimeType.JSON);
