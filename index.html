Configurar Google Sheets
🔗 Integração com Google Sheets + Apps Script
Configure abaixo a URL do seu Google Apps Script (GAS) para salvar os dados automaticamente na planilha. Siga o passo a passo abaixo para ativar a integração.
1. URL do Google Apps Script (Web App)
⚠️ Importante: Ao publicar o Web App no GAS, selecione "Qualquer pessoa" em "Quem tem acesso". Após qualquer alteração no código GAS, clique em "Nova implantação" (não "Gerenciar implantações").
2. Passo a Passo – Configurar o Google Apps Script
1. Acesse script.google.com → clique em Novo projeto
2. Apague o código padrão e cole todo o código abaixo. Substitua SEU_ID_DA_PLANILHA pelo ID da sua planilha Google Sheets
3. Clique em Implantar → Nova implantação → Tipo: App da Web → Executar como: Eu mesmo → Acesso: Qualquer pessoa → Clique em Implantar
4. Copie a URL do Web App gerada e cole no campo acima. Clique em Salvar Configuração e depois Testar Conexão
⚠ Se fizer qualquer mudança no código GAS, sempre use Nova implantação (nunca "Gerenciar implantações existentes")
// ================================================================ // SGQ – Google Apps Script | Cole em: script.google.com // IMPORTANTE: Publicar como Web App → Executar como: Eu mesmo // Quem tem acesso: Qualquer pessoa (sem login) // Após qualquer edição: Nova implantação (não Gerenciar) // ================================================================ const SS_ID = 'SEU_ID_DA_PLANILHA'; // ← Cole aqui o ID da sua planilha // (O ID está na URL: docs.google.com/spreadsheets/d/ID_AQUI/edit) // ---- Headers de cada aba ---- const HEADERS = { AQ: ['numero','data','unidade','departamento','origem','tipoProblema','descricao','custo','tipoCusto','status'], Brainstorm: ['aqNumero','causa'], Ishikawa: ['aqNumero','categoria','causa'], '5Porques': ['aqNumero','nivel','pergunta','resposta'], PlanoAcao: ['aqNumero','acao','responsavel','prazo','status'], Auditorias: ['data','unidade','area','auditor','total','ok','nok','na','score'], Checklist: ['auditoriaId','pergunta','resposta','observacao'], POP: ['tipo','codigo','processo','versao','responsavel','dataRevisao','elaborado','aprovado'], Processos: ['nome','descricao','etapas'], }; function doPost(e) { try { // Accepts text/plain (no CORS preflight) with JSON body const body = e.postData ? e.postData.contents : '{}'; const params = JSON.parse(body); const action = params.action; const data = params.data || {}; const ss = SpreadsheetApp.openById(SS_ID); const actionMap = { saveAQ: ['AQ', data], saveBrainstorm:['Brainstorm', data], saveIshikawa: ['Ishikawa', data], save5Por: ['5Porques', data], savePlano: ['PlanoAcao', data], saveAuditoria: ['Auditorias', data], saveChecklist: ['Checklist', data], savePOP: ['POP', data], saveProcesso: ['Processos', data], }; if (actionMap[action]) { const [sheetName, rowData] = actionMap[action]; upsertRow(ss, sheetName, rowData); } return json({success: true, action}); } catch(err) { return json({success: false, error: err.message}); } } function doGet(e) { try { const action = e.parameter.action || ''; if (action === 'ping') return json({pong: true, ts: new Date().toISOString()}); const sheet = e.parameter.sheet || 'AQ'; const ss = SpreadsheetApp.openById(SS_ID); const sh = ss.getSheetByName(sheet); if (!sh || sh.getLastRow() < 2) return json({data: []}); const rows = sh.getDataRange().getValues(); const headers = rows[0]; const data = rows.slice(1).map(row => { const obj = {}; headers.forEach((h, i) => obj[h] = row[i]); return obj; }); return json({data}); } catch(err) { return json({error: err.message}); } } // Insere nova linha ou atualiza existente (baseado na 1ª coluna = chave) function upsertRow(ss, sheetName, data) { let sh = ss.getSheetByName(sheetName); if (!sh) { sh = ss.insertSheet(sheetName); } const headers = HEADERS[sheetName] || Object.keys(data); // Set headers if sheet is empty if (sh.getLastRow() === 0) { sh.appendRow(headers); sh.getRange(1, 1, 1, headers.length).setFontWeight('bold').setBackground('#e8f0fe'); sh.setFrozenRows(1); } // Build row in header order const row = headers.map(h => { const val = data[h]; return (val === undefined || val === null) ? '' : val; }); // Try to find and update existing row by first key const key = headers[0]; const keyVal = String(data[key] || ''); if (keyVal) { const allData = sh.getDataRange().getValues(); for (let i = 1; i < allData.length; i++) { if (String(allData[i][0]) === keyVal) { sh.getRange(i + 1, 1, 1, row.length).setValues([row]); return {updated: true}; } } } sh.appendRow(row); return {inserted: true}; } function json(obj) { return ContentService .createTextOutput(JSON.stringify(obj)) .setMimeType(ContentService.MimeType.JSON); }
